        % Parameters for size: scale (number) or width (length), height (length)
        % Parameters for the rotation: angle (angle in degrees)
        % To set the width of the text, use width=\textwidth.

      \graphicspath{{figures/}}

    % Figures environment:

        % To center a figure, use the \centering command.
